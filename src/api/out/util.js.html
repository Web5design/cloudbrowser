(function() {
  var Nodemailer, Util;

  Nodemailer = require("nodemailer");

  /**
      @class cloudbrowser.Util
      @param {object} config
  */

  Util = (function() {
    var _instance, _privates;

    _privates = [];

    _instance = null;

    function Util(config) {
      if (_privates.length) {
        return _instance;
      } else {
        _instance = this;
      }
      Object.defineProperty(this, "_index", {
        value: _privates.length
      });
      _privates.push({
        config: config
      });
    }

    /**
        Sends an email to the specified user.
        @static
        @method sendEmail
        @memberOf cloudbrowser.Util
        @param {string} toEmailID
        @param {string} subject
        @param {string} message
        @param {emptyCallback} callback
    */

    Util.prototype.sendEmail = function(toEmailID, subject, message, callback) {
      var mailOptions, smtpTransport;
      smtpTransport = Nodemailer.createTransport("SMTP", {
        service: "Gmail",
        auth: {
          user: _privates[this._index].config.nodeMailerEmailID,
          pass: _privates[this._index].config.nodeMailerPassword
        }
      });
      mailOptions = {
        from: _privates[this._index].config.nodeMailerEmailID,
        to: toEmailID,
        subject: subject,
        html: message
      };
      return smtpTransport.sendMail(mailOptions, function(err, response) {
        if (err) throw err;
        smtpTransport.close();
        return callback();
      });
    };

    return Util;

  })();

  module.exports = Util;

}).call(this);

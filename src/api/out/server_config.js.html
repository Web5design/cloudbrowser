(function() {
  var ServerConfig, compare;

  compare = require('./utils').compare;

  /**
      @class cloudbrowser.ServerConfig
  */

  ServerConfig = (function() {
    var _instance, _privates;

    _privates = [];

    _instance = null;

    function ServerConfig(server) {
      if (_privates.length) {
        return _instance;
      } else {
        _instance = this;
      }
      Object.defineProperty(this, "_index", {
        value: _privates.length
      });
      _privates.push({
        server: server
      });
    }

    /**
        Returns the domain as configured in the server_config.json configuration
        file or as provided through the command line at the time of starting
        CloudBrowser.    
        @static
        @method getDomain
        @memberOf cloudbrowser.ServerConfig
        @return {String}
    */

    ServerConfig.prototype.getDomain = function() {
      return _privates[this._index].server.config.domain;
    };

    /**
        Returns the port as configured in the server_config.json configuration
        file or as provided through the command line at the time of starting
        CloudBrowser.    
        @static
        @method getPort
        @memberOf cloudbrowser.ServerConfig
        @return {Number}
    */

    ServerConfig.prototype.getPort = function() {
      return _privates[this._index].server.config.port;
    };

    /**
        Returns the URL at which the CloudBrowser server is hosted.    
        @static
        @method getUrl
        @memberOf cloudbrowser.ServerConfig
        @return {String}
    */

    ServerConfig.prototype.getUrl = function() {
      return "http://" + this.getDomain() + ":" + this.getPort();
    };

    ServerConfig.prototype.mount = function(path) {
      return _privates[this._index].server.applicationManager.create(path);
    };

    ServerConfig.prototype.unmount = function(mountPoint) {
      return _privates[this._index].server.applicationManager.remove(mountPoint);
    };

    ServerConfig.prototype.listApps = function() {
      var user;
      user = this.getCreator();
      return _privates[this._index].server.applicationManager.get({
        email: user.getEmail(),
        ns: user.getNameSpace()
      });
    };

    /**
        @typedef appObject
        @property {string} mountPoint
        @property {string} description
    */

    /**
        Returns the list of apps mounted on CloudBrowser.    
        @static
        @method getApps
        @memberOf cloudbrowser.ServerConfig
        @return {Array<appObject>}
    */

    ServerConfig.prototype.getApps = function() {
      var app, list, mountPoint, _ref;
      list = [];
      _ref = _privates[this._index].server.applicationManager.get();
      for (mountPoint in _ref) {
        app = _ref[mountPoint];
        list.push({
          mountPoint: mountPoint,
          description: app.description
        });
      }
      list.sort(compare);
      return list;
    };

    ServerConfig.prototype.addEventListener = function(event, callback) {
      return _privates[this._index].server.applicationManager.on(event, function(app) {
        if (event === "Added") {
          return callback({
            mountPoint: app.mountPoint,
            description: app.description
          });
        } else {
          return callback(app.mountPoint);
        }
      });
    };

    return ServerConfig;

  })();

  module.exports = ServerConfig;

}).call(this);

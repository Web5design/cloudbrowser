(function() {
  var Crypto, GoogleStrategy, LocalStrategy, getParentMountPoint, hashPassword, _ref;

  Crypto = require("crypto");

  _ref = require("./utils"), getParentMountPoint = _ref.getParentMountPoint, hashPassword = _ref.hashPassword;

  /**
      @class cloudbrowser.app.LocalStrategy
      @param {BrowserServer} bserver
      @param {cloudbrowser} cloudbrowserContext
  */

  LocalStrategy = (function() {
    var _privates;

    _privates = [];

    function LocalStrategy(bserver, cloudbrowserContext) {
      var appMgr, parentMountPoint;
      Object.defineProperty(this, "_index", {
        value: _privates.length
      });
      parentMountPoint = getParentMountPoint(bserver.mountPoint);
      appMgr = bserver.server.applicationManager;
      _privates.push({
        bserver: bserver,
        parentDbName: appMgr.find(parentMountPoint).dbName,
        browserMgr: appMgr.find(bserver.mountPoint).browsers,
        util: cloudbrowserContext.getUtil(),
        parentMountPoint: parentMountPoint
      });
    }

    /**
        Logs a user into the application.    
        @method login
        @memberof cloudbrowser.app.LocalStrategy
        @instance
        @param options 
        @param {User} options.user
        @param {String} options.password
        @param {booleanCallback} options.callback
    */

    LocalStrategy.prototype.login = function(options) {
      var appUrl, config, mongoInterface,
        _this = this;
      mongoInterface = _privates[this._index].bserver.server.mongoInterface;
      config = _privates[this._index].bserver.server.config;
      appUrl = "http://" + config.domain + ":" + config.port + _privates[this._index].parentMountPoint;
      if (!options.user) {
        throw new Error("Missing required parameter - user");
      } else if (!options.password) {
        throw new Error("Missing required paramter - password");
      }
      return mongoInterface.findUser(options.user.toJson(), _privates[this._index].parentDbName, function(userRec) {
        if (userRec && userRec.status !== 'unverified') {
          return hashPassword({
            password: options.password,
            salt: new Buffer(userRec.salt, 'hex')
          }, function(result) {
            if (result.key.toString('hex') === userRec.key) {
              return _privates[_this._index].bserver.getSessions(function(sessionIDs) {
                return mongoInterface.getSession(sessionIDs[0], function(session) {
                  var redirectto, sessionObj;
                  sessionObj = {
                    app: _privates[_this._index].parentMountPoint,
                    email: options.user.getEmail(),
                    ns: options.user.getNameSpace()
                  };
                  if (!session.user) {
                    session.user = [sessionObj];
                  } else {
                    session.user.push(sessionObj);
                  }
                  redirectto = session.redirectto;
                  session.redirectto = null;
                  return mongoInterface.setSession(sessionIDs[0], session, function() {
                    if (redirectto != null) {
                      _privates[_this._index].bserver.redirect(redirectto);
                    } else {
                      _privates[_this._index].bserver.redirect(appUrl);
                    }
                    return setTimeout(function() {
                      return _privates[_this._index].browserMgr.close(_privates[_this._index].bserver);
                    }, 500);
                  });
                });
              });
            } else if (options.callback) {
              return options.callback(false);
            }
          });
        } else if (options.callback) {
          return options.callback(false);
        }
      });
    };

    /**
        Registers a user with the application and sends a confirmation email to the user's registered email ID.
        The email ID is not activated until it has been confirmed by the user.    
        @memberof cloudbrowser.app.LocalStrategy
        @instance
        @method signup
        @param options 
        @param {User} options.user
        @param {String} options.password
        @param {booleanCallback} options.callback
    */

    LocalStrategy.prototype.signup = function(options) {
      var appUrl, config, mongoInterface,
        _this = this;
      mongoInterface = _privates[this._index].bserver.server.mongoInterface;
      config = _privates[this._index].bserver.server.config;
      appUrl = "http://" + config.domain + ":" + config.port + _privates[this._index].parentMountPoint;
      if (!options.user) {
        throw new Error("Missing required parameter - user");
      } else if (!options.password) {
        throw new Error("Missing required paramter - password");
      }
      return Crypto.randomBytes(32, function(err, token) {
        var confirmationMsg, subject;
        if (err) throw err;
        token = token.toString('hex');
        subject = "Activate your cloudbrowser account";
        confirmationMsg = "Please click on the link below to verify your email address.<br>" + ("<p><a href='" + appUrl + "/activate/" + token + "'>Activate your account</a></p>") + "<p>If you have received this message in error and did not sign up for a cloudbrowser account," + (" click <a href='" + appUrl + "/deactivate/" + token + "'>not my account</a></p>");
        return _privates[_this._index].util.sendEmail(options.user.getEmail(), subject, confirmationMsg, function() {
          return hashPassword({
            password: options.password
          }, function(result) {
            var userRec;
            userRec = {
              email: options.user.getEmail(),
              key: result.key.toString('hex'),
              salt: result.salt.toString('hex'),
              status: 'unverified',
              token: token,
              ns: options.user.getNameSpace()
            };
            return mongoInterface.addUser(userRec, _privates[_this._index].parentDbName, function(user) {
              if (options.callback) return options.callback();
            });
          });
        });
      });
    };

    return LocalStrategy;

  })();

  GoogleStrategy = (function() {
    var _privates;

    _privates = [];

    function GoogleStrategy(bserver) {
      Object.defineProperty(this, "_index", {
        value: _privates.length
      });
      _privates.push({
        parentMountPoint: getParentMountPoint(bserver.mountPoint),
        bserver: bserver,
        browserMgr: bserver.server.applicationManager.find(bserver.mountPoint).browsers
      });
    }

    /**
        Log in through a google ID
        @method login
        @memberof cloudbrowser.app.GoogleStrategy
        @instance
    */

    GoogleStrategy.prototype.login = function() {
      var config, mongoInterface,
        _this = this;
      mongoInterface = _privates[this._index].bserver.server.mongoInterface;
      config = _privates[this._index].bserver.server.config;
      return _privates[this._index].bserver.getSessions(function(sessionIDs) {
        return mongoInterface.getSession(sessionIDs[0], function(session) {
          session.mountPoint = _privates[_this._index].parentMountPoint;
          return mongoInterface.setSession(sessionIDs[0], session, function() {
            _privates[_this._index].bserver.redirect("http://" + config.domain + ":" + config.port + "/googleAuth");
            return setTimeout(function() {
              return _privates[_this._index].browserMgr.close(_privates[_this._index].bserver);
            }, 500);
          });
        });
      });
    };

    /**
        Registers a user with the _application
        @method signup
        @memberof cloudbrowser.app.GoogleStrategy
        @instance
    */

    GoogleStrategy.prototype.signup = GoogleStrategy.prototype.login;

    return GoogleStrategy;

  })();

  module.exports = {

    GoogleStrategy: GoogleStrategy
  };
